\begin{frame}{Why I Studied This Topic}
\begin{itemize}
  \item I studied Paxos.
  \item I wanted to clarify the difference between crash faults and Byzantine faults.
  \item I also wanted to understand why PBFT requires \(3f+1\) replicas.
  \item Therefore, I read ``\href{https://lamport.azurewebsites.net/pubs/byz.pdf}{The Byzantine Generals Problem}'' as theoretical background.
\end{itemize}

\begin{block}{Main Purpose}
My goal was not only to read the paper, but to understand the foundation of Byzantine Fault Tolerance.
\end{block}
\end{frame}

\begin{frame}{My Understanding of Fault Models}
\begin{columns}
\column{0.48\textwidth}
\begin{block}{Crash Fault}
\centering
\begin{tikzpicture}[node distance=1cm]
\node[draw, rounded corners, minimum width=2.2cm] (a) {Node A};
\node[below=of a, text=red] (s) {Stopped};
\draw[-{Latex}] (a) -- (s);
\end{tikzpicture}

\begin{itemize}
  \item Stops responding
  \item Easier to detect
  \item Paxos / Raft
  \item \(2f+1\) replicas
\end{itemize}
\end{block}

\column{0.48\textwidth}
\begin{alertblock}{Byzantine Fault}
\centering
\begin{tikzpicture}[node distance=1cm]
\node[draw, rounded corners, minimum width=2.2cm] (a) {Node A};
\node[below left=of a] (b) {B};
\node[below right=of a] (c) {C};
\draw[-{Latex}] (a) -- node[left] {X} (b);
\draw[-{Latex}] (a) -- node[right] {Y} (c);
\end{tikzpicture}

\begin{itemize}
  \item Behaves arbitrarily
  \item Sends conflicting messages
  \item PBFT / BFT protocols
  \item \(3f+1\) replicas
\end{itemize}
\end{alertblock}
\end{columns}
\end{frame}

\begin{frame}{The Core Problem I Understood}
\begin{center}
\begin{tikzpicture}[node distance=1.6cm]
\node[draw, rounded corners, fill=red!15] (faulty) {Faulty Node};
\node[draw, rounded corners, below left=of faulty] (b) {Correct Node B};
\node[draw, rounded corners, below right=of faulty] (c) {Correct Node C};

\draw[-{Latex}, thick] (faulty) -- node[left] {value X} (b);
\draw[-{Latex}, thick] (faulty) -- node[right] {value Y} (c);

\node[below=1.1cm of faulty] (problem) {\Large Conflicting local views};
\end{tikzpicture}
\end{center}

\begin{alertblock}{Key Insight}
The main problem is not identifying the faulty node.  
The main problem is guaranteeing that correct nodes still make the same decision.
\end{alertblock}
\end{frame}

\begin{frame}{My Understanding of the \(3f+1\) Requirement}
\begin{columns}
\column{0.45\textwidth}
\begin{block}{For \(f=1\)}
\begin{itemize}
  \item 3 nodes are not enough.
  \item 4 nodes are required.
  \item Correct nodes need enough independent reports.
\end{itemize}
\end{block}

\column{0.5\textwidth}
\begin{table}
\centering
\begin{tabular}{cc}
\toprule
Byzantine Faults & Required Nodes \\
\midrule
0 & 1 \\
1 & 4 \\
2 & 7 \\
3 & 10 \\
\bottomrule
\end{tabular}
\end{table}
\end{columns}

\vspace{0.4cm}

\begin{alertblock}{Important Realization}
PBFT's \(3f+1\) requirement is not just an implementation choice.  
It comes from a fundamental impossibility result.
\end{alertblock}
\end{frame}

\begin{frame}{Why 3 Nodes Are Not Enough, But 4 Nodes Are}
\small
\begin{columns}

\column{0.48\textwidth}
\begin{alertblock}{3 Nodes: Ambiguous}
\centering
\begin{tikzpicture}[node distance=1.35cm, scale=0.9, transform shape]
\node[draw, circle, fill=red!15] (a) {A};
\node[draw, circle, below left=of a] (b) {B};
\node[draw, circle, below right=of a] (c) {C};

\draw[-{Latex}, thick] (a) -- node[left] {X} (b);
\draw[-{Latex}, thick] (a) -- node[right] {Y} (c);
\draw[<->, dashed] (b) -- node[below] {compare} (c);
\end{tikzpicture}

\vspace{0.1cm}
\small
B sees X from A and Y from C.  
It cannot know who is faulty.
\end{alertblock}

\column{0.48\textwidth}
\begin{block}{4 Nodes: More Evidence}
\centering
\begin{tikzpicture}[node distance=1.3cm, scale=0.9, transform shape]
\node[draw, circle, fill=red!15] (a) {A};
\node[draw, circle, below left=of a] (b) {B};
\node[draw, circle, below=of a] (c) {C};
\node[draw, circle, below right=of a] (d) {D};

\draw[-{Latex}, thick] (a) -- node[left] {X} (b);
\draw[-{Latex}, thick] (a) -- node[right] {Y} (c);
\draw[-{Latex}, thick] (a) -- node[right] {Y} (d);

\draw[<->, dashed] (b) -- (c);
\draw[<->, dashed] (c) -- (d);
\draw[<->, dashed] (b) to[bend right=20] (d);
\end{tikzpicture}

\vspace{0.1cm}
\small
Correct nodes can compare reports from more independent nodes.
\end{block}

\end{columns}

\vspace{0.15cm}

\begin{block}{My Understanding}
With only 3 nodes, one Byzantine node can create an indistinguishable conflict.  
With 4 nodes, correct nodes have enough additional reports to overcome one faulty node.
\end{block}
\end{frame}

\begin{frame}{What I Understand Now}
\begin{columns}
\column{0.48\textwidth}
\begin{block}{Current Understanding}
\begin{itemize}
  \item Byzantine faults are stronger than crash faults.
  \item The goal is agreement among correct nodes.
  \item Identifying the faulty node is not the primary goal.
  \item \(3f+1\) replicas are necessary.
\end{itemize}
\end{block}

\column{0.48\textwidth}
\begin{block}{PBFT Intuition}
\begin{itemize}
  \item Do not trust the leader alone.
  \item Share proposals among replicas.
  \item Require enough matching messages.
  \item Preserve safety despite faulty replicas.
\end{itemize}
\end{block}
\end{columns}
\end{frame}
